141static void TareaMedir (void *pvParameter)
142{
143 while(true){//prender_sistema==true
144 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
145
146 AnalogInputReadSingle(CH2,&luz_interior);
147//---------------VER SI ESTO FUNCIONA EN LA UART-----------------------
148 UartSendString(UART_PC, "Luz en el interior: ");
149 UartSendString(UART_PC, (char *)UartItoa(luz_interior, 10));
150 UartSendString(UART_PC, "\r\n");
151
152 AnalogInputReadSingle(CH3,&luz_exterior);
153 if (ventanas_abiertas==1){
154 if(luz_interior<iluminacion_ideal){
155 if(luz_exterior>iluminacion_ideal){
156 prender_luces=1;
157 GPIOOn(GPIO_9);
158 }
159 else {
160 if(luz_exterior<iluminacion_ideal){
161 ventanas_abiertas=0;
162 }
163 }
164 }
165 else{
166 iluminacion_optima=1;
167 }
168 }
169 else{
170 if(ventanas_abiertas==0){
171 if(luz_exterior>iluminacion_ideal){
172 GPIOOff(GPIO_9);
173 ventanas_abiertas=1;
174 }
175 if(luz_exterior<iluminacion_ideal) {
176 prender_luces=1;
177 GPIOOn(GPIO_9);
178 }
179 }
180 }
181 }
